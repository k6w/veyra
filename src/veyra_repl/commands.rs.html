5/// Handle REPL commands (starting with :)
6pub fn handle_command(cmd: &str, state: &mut ReplState, ui: &mut UI) -> Result<bool> {
7    let parts: Vec<&str> = cmd.trim().split_whitespace().collect();
8
9    if parts.is_empty() {
10        return Ok(true);
11    }
12
13    match parts[0] {
14        ":help" | ":h" => {
15            show_help(ui);
16        }
17        ":exit" | ":quit" | ":q" => {
18            ui.success("Goodbye! ðŸ‘‹");
19            return Ok(false);
20        }
21        ":clear" | ":cls" => {
22            clear_screen();
23        }
24        ":reset" => {
25            state.reset();
26            ui.success("REPL state has been reset");
27        }
28        ":history" => {
29            show_history(state, ui);
30        }
31        ":vars" | ":variables" => {
32            show_variables(state, ui);
33        }
34        ":funcs" | ":functions" => {
35            show_functions(state, ui);
36        }
37        ":info" => {
38            show_info(state, ui);
39        }
40        ":config" => {
41            if parts.len() > 1 {
42                handle_config_command(&parts[1..], state, ui)?;
43            } else {
44                show_config(state, ui);
45            }
46        }
47        ":theme" => {
48            if parts.len() > 1 {
49                change_theme(parts[1], state, ui)?;
50            } else {
51                ui.info("Available themes: default, monokai, dracula, nord, solarized-dark, solarized-light");
52            }
53        }
54        ":save" => {
55            if parts.len() > 1 {
56                save_session(parts[1], state, ui)?;
57            } else {
58                ui.error("Usage: :save <filename>");
59            }
60        }
61        ":load" => {
62            if parts.len() > 1 {
63                load_file(parts[1], state, ui)?;
64            } else {
65                ui.error("Usage: :load <filename>");
66            }
67        }
68        ":time" => {
69            toggle_timing(state, ui);
70        }
71        ":verbose" => {
72            toggle_verbose(state, ui);
73        }
74        ":multiline" => {
75            toggle_multiline(state, ui);
76        }
77        ":type" => {
78            if parts.len() > 1 {
79                show_type(&parts[1..].join(" "), state, ui)?;
80            } else {
81                ui.error("Usage: :type <expression>");
82            }
83        }
84        ":tips" => {
85            show_tips(ui);
86        }
87        ":themes" => {
88            ui.section("Available Themes");
89            let themes = vec![
90                ("default", "Classic terminal colors"),
91                ("monokai", "Monokai inspired dark theme"),
92                ("dracula", "Dracula theme with vibrant colors"),
93                ("nord", "Nord arctic-inspired palette"),
94                ("solarized-dark", "Solarized dark theme"),
95                ("solarized-light", "Solarized light theme"),
96            ];
97            for (name, desc) in themes {
98                println!("  {} - {}", ui.theme.primary.paint(name), desc);
99            }
100            println!();
101            ui.info("Use ':theme <name>' to switch themes");
102        }
103        _ => {
104            ui.error(&format!("Unknown command: {}", parts[0]));
105            ui.info("Type :help for available commands");
106        }
107    }
108
109    Ok(true)
110}

156fn show_history(state: &ReplState, ui: &UI) {
157    ui.section("Command History");
158
159    let history = state.history();
160    if history.is_empty() {
161        ui.info("No history yet");
162        return;
163    }
164
165    for (i, cmd) in history.iter().rev().take(50).enumerate() {
166        let num = history.len() - i;
167        println!("{:4} â”‚ {}", ui.theme.muted.paint(num.to_string()), cmd);
168    }
169
170    if history.len() > 50 {
171        println!();
172        ui.info(&format!("Showing last 50 of {} entries", history.len()));
173    }
174}

311fn handle_config_command(args: &[&str], state: &mut ReplState, ui: &UI) -> Result<()> {
312    if args.is_empty() {
313        show_config(state, ui);
314        return Ok(());
315    }
316
317    match args[0] {
318        "set" => {
319            if args.len() < 3 {
320                ui.error("Usage: :config set <key> <value>");
321                return Ok(());
322            }
323
324            let key = args[1];
325            let value = args[2];
326
327            let config = state.config_mut();
328
329            match key {
330                "syntax_highlighting" => {
331                    config.syntax_highlighting = value.parse().unwrap_or(true);
332                    ui.success(&format!("Set syntax_highlighting to {}", value));
333                }
334                "auto_completion" => {
335                    config.auto_completion = value.parse().unwrap_or(true);
336                    ui.success(&format!("Set auto_completion to {}", value));
337                }
338                "multiline_mode" => {
339                    config.multiline_mode = value.parse().unwrap_or(true);
340                    ui.success(&format!("Set multiline_mode to {}", value));
341                }
342                "show_timing" => {
343                    config.show_timing = value.parse().unwrap_or(true);
344                    ui.success(&format!("Set show_timing to {}", value));
345                }
346                "vi_mode" => {
347                    config.vi_mode = value.parse().unwrap_or(false);
348                    ui.success(&format!("Set vi_mode to {}", value));
349                    ui.warning("Restart REPL for this change to take effect");
350                }
351                "show_tips" => {
352                    config.show_tips = value.parse().unwrap_or(true);
353                    ui.success(&format!("Set show_tips to {}", value));
354                }
355                "auto_indent" => {
356                    config.auto_indent = value.parse().unwrap_or(true);
357                    ui.success(&format!("Set auto_indent to {}", value));
358                }
359                "auto_insert_function_parens" => {
360                    config.auto_insert_function_parens = value.parse().unwrap_or(true);
361                    ui.success(&format!("Set auto_insert_function_parens to {}", value));
362                }
363                "fancy_errors" => {
364                    config.fancy_errors = value.parse().unwrap_or(true);
365                    ui.success(&format!("Set fancy_errors to {}", value));
366                }
367                _ => {
368                    ui.error(&format!("Unknown config key: {}", key));
369                }
370            }
371
372            config.save()?;
373        }
374        "save" => {
375            state.config().save()?;
376            ui.success("Configuration saved");
377        }
378        _ => {
379            ui.error(&format!("Unknown config command: {}", args[0]));
380        }
381    }
382
383    Ok(())
384}

386fn change_theme(theme_name: &str, state: &mut ReplState, ui: &mut UI) -> Result<()> {
387    let new_theme = match theme_name.to_lowercase().as_str() {
388        "default" => Some((Theme::default(), crate::config::ColorScheme::Default)),
389        "monokai" => Some((Theme::monokai(), crate::config::ColorScheme::Monokai)),
390        "dracula" => Some((Theme::dracula(), crate::config::ColorScheme::Dracula)),
391        "nord" => Some((Theme::nord(), crate::config::ColorScheme::Nord)),
392        "solarized-dark" => Some((
393            Theme::solarized_dark(),
394            crate::config::ColorScheme::SolarizedDark,
395        )),
396        "solarized-light" => Some((
397            Theme::solarized_light(),
398            crate::config::ColorScheme::SolarizedLight,
399        )),
400        _ => None,
401    };
402
403    if let Some((theme, scheme)) = new_theme {
404        ui.theme = theme;
405        state.config_mut().color_scheme = scheme;
406        state.config().save()?;
407        ui.success(&format!("Theme changed to '{}'", theme_name));
408    } else {
409        ui.error(&format!("Unknown theme: '{}'", theme_name));
410        ui.info(
411            "Available themes: default, monokai, dracula, nord, solarized-dark, solarized-light",
412        );
413    }
414    Ok(())
415}

429fn load_file(filename: &str, state: &mut ReplState, ui: &UI) -> Result<()> {
430    use crate::ui::with_spinner;
431
432    let path = std::path::Path::new(filename);
433
434    if !path.exists() {
435        ui.error(&format!("File not found: {}", filename));
436        return Ok(());
437    }
438
439    with_spinner(&format!("Loading {}...", filename), || {
440        state.load_file(path)
441    })?;
442
443    ui.success(&format!("Loaded {}", filename));
444    Ok(())
445}

473fn show_type(expr: &str, state: &mut ReplState, ui: &UI) -> Result<()> {
474    match state.execute(expr) {
475        Ok(Some(value)) => {
476            let type_name = crate::state::type_name(&value);
477            ui.info(&format!("Type: {}", type_name));
478        }
479        Ok(None) => {
480            ui.warning("Expression returned no value");
481        }
482        Err(e) => {
483            ui.error(&format!("Error evaluating expression: {}", e));
484        }
485    }
486    Ok(())
487}
